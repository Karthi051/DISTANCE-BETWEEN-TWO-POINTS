  ```
  #Program to find the distance between two points.
#Developed by: karthi keyan k

import math 
x1,y1=10,6
x2,y2=4,2
d=math.sqrt((x2-x1)**2+(y2-y1)**2)
e="{:.2f}".format(d)
print(e)
  ```
